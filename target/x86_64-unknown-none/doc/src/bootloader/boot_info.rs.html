3/// This structure represents the information that the bootloader passes to the kernel.
4///
5/// The information is passed as an argument to the entry point. The entry point function must
6/// have the following signature:
7///
8/// ```
9/// # use bootloader::BootInfo;
10/// # type _SIGNATURE =
11/// extern "C" fn(boot_info: &'static mut BootInfo) -> !;
12/// ```
13///
14/// Note that no type checking occurs for the entry point function, so be careful to
15/// use the correct argument types. To ensure that the entry point function has the correct
16/// signature, use the [`entry_point`] macro.
17#[derive(Debug)]
18#[repr(C)]
19#[non_exhaustive]
20pub struct BootInfo {
21    /// Bootloader version (major).
22    pub version_major: u16,
23    /// Bootloader version (minor).
24    pub version_minor: u16,
25    /// Bootloader version (patch).
26    pub version_patch: u16,
27    /// Whether the bootloader version is a pre-release.
28    ///
29    /// We can't store the full prerelease string of the version number since it could be
30    /// arbitrarily long.
31    pub pre_release: bool,
32    /// A map of the physical memory regions of the underlying machine.
33    ///
34    /// The bootloader queries this information from the BIOS/UEFI firmware and translates this
35    /// information to Rust types. It also marks any memory regions that the bootloader uses in
36    /// the memory map before passing it to the kernel. Regions marked as usable can be freely
37    /// used by the kernel.
38    pub memory_regions: MemoryRegions,
39    /// Information about the framebuffer for screen output if available.
40    pub framebuffer: Optional<FrameBuffer>,
41    /// The virtual address at which the mapping of the physical memory starts.
42    ///
43    /// Physical addresses can be converted to virtual addresses by adding this offset to them.
44    ///
45    /// The mapping of the physical memory allows to access arbitrary physical frames. Accessing
46    /// frames that are also mapped at other virtual addresses can easily break memory safety and
47    /// cause undefined behavior. Only frames reported as `USABLE` by the memory map in the `BootInfo`
48    /// can be safely accessed.
49    ///
50    /// Only available if the `map-physical-memory` config option is enabled.
51    pub physical_memory_offset: Optional<u64>,
52    /// The virtual address of the recursively mapped level 4 page table.
53    ///
54    /// Only available if the `map-page-table-recursively` config option is enabled.
55    pub recursive_index: Optional<u16>,
56    /// The address of the `RSDP` data structure, which can be use to find the ACPI tables.
57    ///
58    /// This field is `None` if no `RSDP` was found (for BIOS) or reported (for UEFI).
59    pub rsdp_addr: Optional<u64>,
60    /// The thread local storage (TLS) template of the kernel executable, if present.
61    pub tls_template: Optional<TlsTemplate>,
62}

64/// FFI-safe slice of [`MemoryRegion`] structs, semantically equivalent to
65/// `&'static mut [MemoryRegion]`.
66///
67/// This type implements the [`Deref`][core::ops::Deref] and [`DerefMut`][core::ops::DerefMut]
68/// traits, so it can be used like a `&mut [MemoryRegion]` slice. It also implements [`From`]
69/// and [`Into`] for easy conversions from and to `&'static mut [MemoryRegion]`.
70#[derive(Debug)]
71#[repr(C)]
72pub struct MemoryRegions {
73    pub(crate) ptr: *mut MemoryRegion,
74    pub(crate) len: usize,
75}
76
77impl ops::Deref for MemoryRegions {
78    type Target = [MemoryRegion];
79
80    fn deref(&self) -> &Self::Target {
81        unsafe { slice::from_raw_parts(self.ptr, self.len) }
82    }
83}
84
85impl ops::DerefMut for MemoryRegions {
86    fn deref_mut(&mut self) -> &mut Self::Target {
87        unsafe { slice::from_raw_parts_mut(self.ptr, self.len) }
88    }
89}
90
91impl From<&'static mut [MemoryRegion]> for MemoryRegions {
92    fn from(regions: &'static mut [MemoryRegion]) -> Self {
93        MemoryRegions {
94            ptr: regions.as_mut_ptr(),
95            len: regions.len(),
96        }
97    }
98}
99
100impl From<MemoryRegions> for &'static mut [MemoryRegion] {
101    fn from(regions: MemoryRegions) -> &'static mut [MemoryRegion] {
102        unsafe { slice::from_raw_parts_mut(regions.ptr, regions.len) }
103    }
104}

106/// Represent a physical memory region.
107#[derive(Debug, Copy, Clone, Eq, PartialEq)]
108#[repr(C)]
109pub struct MemoryRegion {
110    /// The physical start address of the region.
111    pub start: u64,
112    /// The physical end address (exclusive) of the region.
113    pub end: u64,
114    /// The memory type of the memory region.
115    ///
116    /// Only [`Usable`][MemoryRegionKind::Usable] regions can be freely used.
117    pub kind: MemoryRegionKind,
118}
119
120impl MemoryRegion {
121    /// Creates a new empty memory region (with length 0).
122    pub const fn empty() -> Self {
123        MemoryRegion {
124            start: 0,
125            end: 0,
126            kind: MemoryRegionKind::Bootloader,
127        }
128    }
129}

131/// Represents the different types of memory.
132#[derive(Debug, Copy, Clone, Eq, PartialEq)]
133#[non_exhaustive]
134#[repr(C)]
135pub enum MemoryRegionKind {
136    /// Unused conventional memory, can be used by the kernel.
137    Usable,
138    /// Memory mappings created by the bootloader, including the kernel and boot info mappings.
139    ///
140    /// This memory should _not_ be used by the kernel.
141    Bootloader,
142    /// An unknown memory region reported by the UEFI firmware.
143    ///
144    /// This should only be used if the UEFI memory type is known as usable.
145    UnknownUefi(u32),
146    /// An unknown memory region reported by the BIOS firmware.
147    ///
148    /// This should only be used if the BIOS memory type is known as usable.
149    UnknownBios(u32),
150}

152/// A pixel-based framebuffer that controls the screen output.
153#[derive(Debug)]
154#[repr(C)]
155pub struct FrameBuffer {
156    pub(crate) buffer_start: u64,
157    pub(crate) buffer_byte_len: usize,
158    pub(crate) info: FrameBufferInfo,
159}
160
161impl FrameBuffer {
162    /// Returns the raw bytes of the framebuffer as slice.
163    pub fn buffer(&self) -> &[u8] {
164        unsafe { self.create_buffer() }
165    }
166
167    /// Returns the raw bytes of the framebuffer as mutable slice.
168    pub fn buffer_mut(&mut self) -> &mut [u8] {
169        unsafe { self.create_buffer() }
170    }
171
172    unsafe fn create_buffer<'a>(&self) -> &'a mut [u8] {
173        unsafe { slice::from_raw_parts_mut(self.buffer_start as *mut u8, self.buffer_byte_len) }
174    }
175
176    /// Returns layout and pixel format information of the framebuffer.
177    pub fn info(&self) -> FrameBufferInfo {
178        self.info
179    }
180}

182/// Describes the layout and pixel format of a framebuffer.
183#[derive(Debug, Clone, Copy)]
184#[repr(C)]
185pub struct FrameBufferInfo {
186    /// The total size in bytes.
187    pub byte_len: usize,
188    /// The width in pixels.
189    pub horizontal_resolution: usize,
190    /// The height in pixels.
191    pub vertical_resolution: usize,
192    /// The color format of each pixel.
193    pub pixel_format: PixelFormat,
194    /// The number of bytes per pixel.
195    pub bytes_per_pixel: usize,
196    /// Number of pixels between the start of a line and the start of the next.
197    ///
198    /// Some framebuffers use additional padding at the end of a line, so this
199    /// value might be larger than `horizontal_resolution`. It is
200    /// therefore recommended to use this field for calculating the start address of a line.
201    pub stride: usize,
202}

226/// Information about the thread local storage (TLS) template.
227///
228/// This template can be used to set up thread local storage for threads. For
229/// each thread, a new memory location of size `mem_size` must be initialized.
230/// Then the first `file_size` bytes of this template needs to be copied to the
231/// location. The additional `mem_size - file_size` bytes must be initialized with
232/// zero.
233#[derive(Debug, Clone, Copy, PartialEq, Eq)]
234#[repr(C)]
235pub struct TlsTemplate {
236    /// The virtual start address of the thread local storage template.
237    pub start_addr: u64,
238    /// The number of data bytes in the template.
239    ///
240    /// Corresponds to the length of the `.tdata` section.
241    pub file_size: u64,
242    /// The total number of bytes that the TLS segment should have in memory.
243    ///
244    /// Corresponds to the combined length of the `.tdata` and `.tbss` sections.
245    pub mem_size: u64,
246}

248/// FFI-safe variant of [`Option`].
249///
250/// Implements the [`From`] and [`Into`] traits for easy conversion to and from [`Option`].
251#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash)]
252#[repr(C)]
253pub enum Optional<T> {
254    /// Some value `T`
255    Some(T),
256    /// No value
257    None,
258}
259
260impl<T> Optional<T> {
261    /// Converts the `Optional` to an [`Option`].
262    pub fn into_option(self) -> Option<T> {
263        self.into()
264    }
265
266    /// Converts from `&Optional<T>` to `Option<&T>`.
267    ///
268    /// For convenience, this method directly performs the conversion to the standard
269    /// [`Option`] type.
270    pub const fn as_ref(&self) -> Option<&T> {
271        match self {
272            Self::Some(x) => Some(x),
273            Self::None => None,
274        }
275    }
276
277    /// Converts from `&mut Optional<T>` to `Option<&mut T>`.
278    ///
279    /// For convenience, this method directly performs the conversion to the standard
280    /// [`Option`] type.
281    pub fn as_mut(&mut self) -> Option<&mut T> {
282        match self {
283            Self::Some(x) => Some(x),
284            Self::None => None,
285        }
286    }
287}
288
289impl<T> From<Option<T>> for Optional<T> {
290    fn from(v: Option<T>) -> Self {
291        match v {
292            Some(v) => Optional::Some(v),
293            None => Optional::None,
294        }
295    }
296}
297
298impl<T> From<Optional<T>> for Option<T> {
299    fn from(optional: Optional<T>) -> Option<T> {
300        match optional {
301            Optional::Some(v) => Some(v),
302            Optional::None => None,
303        }
304    }
305}
